era_1_start = { value = 0 }

era_2_start = {
	if = {
		limit = {
			era_2_started_by_variable = 1
		}
		if = {
			limit = {
				exists = global_var:ds_era_2_start
			}
			value = global_var:ds_era_2_start
			add = era_1_start
		}
		else = {
			value = 100000
		}
	}
	else = {
		value = era_1_end
	}
}
